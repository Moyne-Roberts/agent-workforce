    **4. Known-exceptions loader** `web/lib/automations/uren-controle/known-exceptions.ts`:
    ```typescript
    import { createAdminClient } from "@/lib/supabase/admin";
    import type { KnownException, RuleType } from "./types";

    export async function loadKnownExceptions(): Promise<KnownException[]> {
      const admin = createAdminClient();
      const { data, error } = await admin
        .from("known_exceptions")
        .select("employee_name, rule_type, reason")
        .eq("automation", "uren-controle")
        .eq("active", true);
      if (error) throw new Error(`known_exceptions load: ${error.message}`);
      return (data ?? []).map(r => ({
        employeeName: r.employee_name,
        ruleType: r.rule_type as RuleType,
        reason: r.reason,
      }));
    }

    export function shouldSuppress(
      exceptions: KnownException[],
      employeeName: string,
      ruleType: RuleType,
    ): boolean {
      return exceptions.some(e =>
        e.employeeName.toLowerCase() === employeeName.toLowerCase() &&
        e.ruleType === ruleType
      );
    ```

    ```typescript
    // v1 NARROWED CONTRACT (checker W6):
    // Detect rows where the verzuim column(s) voor dezelfde medewerker + dezelfde dag
    // BOTH a 'ziekte' indicator AND a 'verlof'/'vakantie' indicator bevatten.
    // In valide data is dat onmogelijk (je bent ziek OF met verlof, niet beide),
    // dus dit signaleert een BCS dual-registration.
    //
    // Heuristiek: check `day.verzuim` string op trefwoorden (case-insensitive):
    //   heeft 'ziek' substring  AND (heeft 'verlof' OR 'vakantie' OR 'atv') substring → flag
    // Alternatief (als fixture laat zien dat er meerdere verzuim rijen per dag zijn in Mutaties tab):
    //   group Mutaties rows by (employee, date), flag als set{kind} zowel 'ziekte' als een verlof-variant bevat.
    //
    // Als de echte BCS duplicate signature in productie afwijkt van deze heuristiek,
    // is dit een v2 refinement — documenteer in README "Known limitations" en open ticket.
    export function detectVerzuimBcsDuplicate(employee: EmployeeData): FlaggedRow[] {
      const flags: FlaggedRow[] = [];
      for (const day of employee.days) {
        const v = (day.verzuim ?? '').toLowerCase();
        if (!v) continue;
        const hasSick = v.includes('ziek');
        const hasLeave = v.includes('verlof') || v.includes('vakantie') || v.includes('atv');
        if (hasSick && hasLeave) {
          flags.push({
            employeeName: employee.name,
            employeeCategory: employee.category,
            ruleType: 'verzuim_bcs_duplicate',
            severity: 'review',
            dayDate: day.date,
            weekNumber: null,
            rawValues: { verzuim: day.verzuim },
            description: `Verzuim dual-registration op ${day.date}: bevat zowel ziekte als verlof indicator.`,
          });
        }
      }
      return flags;
    }

    export function runAllRules(
      parsed: ParsedHourCalculation,
      exceptions: KnownException[],
    ): FlaggedRow[] {
      const flags: FlaggedRow[] = [];
      for (const emp of parsed.employees) {
        for (const day of emp.days) {
          const tnt = detectTnTMismatch(day, emp);
          if (tnt) flags.push(tnt);
          const outlier = detectVerschilOutlier(day, emp);
          if (outlier) flags.push(outlier);
        }
        flags.push(...detectWeekendFlip(emp));
        flags.push(...detectVerzuimBcsDuplicate(emp));
      }
      // Exceptions don't remove flags here; caller tags suppressed_by_exception in DB insert.
      return flags;
    }

    export function isSuppressed(flag: FlaggedRow, exceptions: KnownException[]): boolean {
      return shouldSuppress(exceptions, flag.employeeName, flag.ruleType);
    ```
